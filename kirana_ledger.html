CREATE TABLE customers ( id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, phone TEXT, address TEXT, created_at TEXT DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE transactions ( id INTEGER PRIMARY KEY AUTOINCREMENT, customer_id INTEGER REFERENCES customers(id), type TEXT CHECK(type IN ('credit','payment')), amount REAL NOT NULL, description TEXT, date TEXT, due_date TEXT, created_at TEXT DEFAULT CURRENT_TIMESTAMP );
